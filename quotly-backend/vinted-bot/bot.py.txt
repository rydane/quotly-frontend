import asyncio
import aiohttp
from bs4 import BeautifulSoup
import webbrowser

# 🔔 TELEGRAM
TOKEN = "TON_TOKEN"
CHAT_ID = "TON_CHAT_ID"

BASE_URL = "https://www.vinted.co.uk"

# 🎯 KEYWORDS NICHE RUNNING
KEYWORDS = [
    "nike elite", "nike phenom", "nike phénome", "nike division",
    "nike pants", "nike running", "nike dri-fit",
    "nike aeroswift", "nike pegasus", "nike zoom",
    "nike trail", "nike pro", "nike tights", "nike jogger"
]

# 🔍 RECHERCHE
SEARCH_URL = (
    f"{BASE_URL}/catalog?"
    "search_text=nike"
    "&price_to=40"
    "&order=newest_first"
)

HEADERS = {
    "User-Agent": "Mozilla/5.0",
    "Connection": "keep-alive"
}

seen_items = set()

# 💰 PRIX REVENTE ESTIMÉS (VINTED FR)
RESALE_PRICES_FR = {
    "nike elite": 45,
    "nike phenom": 35,
    "nike aeroswift": 50,
    "nike pegasus": 60,
    "nike zoom": 50
}

# 🧠 FILTRE RAPIDE
def is_relevant(title):
    t = title.lower()
    return "nike" in t and any(k in t for k in KEYWORDS)

# 💰 EXTRACTION PRIX
def extract_price(p):
    return float(p.replace("£", "").replace(",", ".").strip())

# 📈 ESTIMATION REVENTE
def estimate_resale(title):
    t = title.lower()
    for k, v in RESALE_PRICES_FR.items():
        if k in t:
            return v
    return None

# 💸 CALCUL PROFIT
def calculate_profit(title, buy_price):
    resale = estimate_resale(title)
    if resale:
        return resale - buy_price, resale
    return None, None

# 🔔 TELEGRAM
async def send_telegram(session, msg):
    await session.post(
        f"https://api.telegram.org/bot{TOKEN}/sendMessage",
        data={
            "chat_id": CHAT_ID,
            "text": msg,
            "parse_mode": "Markdown"
        }
    )

# 🌐 FETCH
async def fetch(session, url):
    async with session.get(url, headers=HEADERS) as r:
        return await r.text()

# 🔍 SCAN PAGE
async def scan(session, page):
    url = SEARCH_URL + f"&page={page}"
    html = await fetch(session, url)
    soup = BeautifulSoup(html, "lxml")
    return soup.select(".feed-grid__item")

# 🚀 MAIN BOT
async def main():
    connector = aiohttp.TCPConnector(limit=200)

    async with aiohttp.ClientSession(connector=connector) as session:
        while True:
            tasks = [scan(session, p) for p in range(1, 7)]
            results = await asyncio.gather(*tasks)

            for items in results:
                for item in items:
                    try:
                        title = item.select_one(".feed-grid__item-title").text.strip()
                        link = BASE_URL + item.get("href")

                        if link in seen_items:
                            continue

                        if not is_relevant(title):
                            continue

                        price_text = item.select_one(".feed-grid__item-price").text.strip()
                        price = extract_price(price_text)

                        seen_items.add(link)

                        profit, resale = calculate_profit(title, price)

                        # 🎯 FILTRE PROFIT (minimum 15€)
                        if profit and profit >= 15:
                            msg = f"""
💣 *STEAL DÉTECTÉ*

👟 {title}
💰 {price_text}

📈 Revente estimée : {resale}€
💸 Profit : +{profit}€

⚡ ACHETER :
{link}
"""
                            await send_telegram(session, msg)

                            # ⚡ SEMI AUTO COP (ouvre direct)
                            webbrowser.open(link)

                    except:
                        continue

            await asyncio.sleep(2)  # ⚡ ultra rapide

asyncio.run(main())
